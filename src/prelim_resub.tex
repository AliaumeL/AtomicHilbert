\section{Background}

\subsection{Polynomial ideals}
%
Let $\K$ be an arbitrary field and $X$ be a set of variables.
We use $\poly{\K}{X}$ denote the polynomial ring over $\K$ in these variables.

A subset $\idl[I] \subseteq \poly{\K}{X}$ is an \intro{ideal} if
\begin{enumerate}
\item $0 \in \idl[I]$,
\item for every $f,g$ in $\idl[I]$, their sum $f + g$ is also in $\idl[I]$, and
\item for every $f\in\idl[I]$ and $h\in\poly{\K}{X}$, their product $f \cdot h$ is also in $\idl[I]$.
\end{enumerate}
%
\begin{example}\label{eg:I0}
Let $\idl[I]_0^X$ be the set of polynomials whose constant term is $0$.
For illustration,
the polynomial $x^2 + y$ is in $\idl[I]_0^{\{x,y\}}$ and the polynomial $xy + 2$ is not.
We leave it to the reader to check that $\idl[I]_0^X$ is indeed an ideal. 
\end{example}
%
\arka{possible repetition}.
%
\begin{example}\label{eg:I1}
Let $\idl[I]_1^X$ be the set of polynomials whose coefficients add up to $1$.
For illustration,
the polynomial $x^2 - 2y + 1$ is in $\idl[I]_0^{\{x,y\}}$ and the polynomial $xy + y^2$ is not. 
\end{example}
%
For a subset $F \subseteq \poly{\K}{X}$,
the ideal \intro{$\IdlGen{F}$} generated by $H$ is the smallest ideal containing $F$.
%
\begin{remark}
The ideal $\IdlGen{F}$ generated by $F\subseteq \poly{\K}{X}$ can equivalently defined as the set of all polynomials that can be written as
$
\sum_{i = 1}^n h_i \cdot f_i
$
for $h_i\in\poly{\K}{X}$ and $f_i \in F$.
\end{remark}
%
\begin{example}
The ideals $\idl[I]_0$ and $\idl[I]_1$ defined in \Cref{eg:I0,eg:I1} are generated respectively by the sets $\setof{x}{x\in X}$ and $\setof{x - 1}{x\in X}$.
\end{example}

\subsection{Polynomial rings with finitely many variables}\label{sec:polyfin}
%
An ideal $\idl[I]$ is called \intro{finitely generated} if it is generated by a finite set of polynomials.
For instance, when $X$ is finite, the ideals $\idl[I]_0^X$ and $\idl[I]_1^X$ are finitely generated.
In fact,
%
\begin{theorem}[Hilbert's basis theorem]
If $X$ is finite,
then every ideal in $\poly{\K}{X}$ is finitely generated.
\end{theorem}
%
In the rest of this subsection we will assume that the set of variables $X$ is finite.

Hilbert's basis theorem implies that ideals in $\poly{\K}{X}$ can be finitely represented (assuming elements of the field $\K$ can also be finitely represented)and given as input to algorithms .
One of the most important decision problem regarding ideals is the \intro{ideal membership problem}:
decide whether a given polynomial $f\in\poly{\K}{X}$ is in the ideal generated by a given finite set of polynomials $P\subseteq\poly{\K}{X}$.

The usual procedure to decide ideal membership is to first compute a \kl{Gr\"{o}bner basis} $G$ of $\IdlGen{P}$ using \kl{Buchberger's algorithm},
divide $f$ by $G$ using the \kl{multivariate division algorithm},
if result of the division is $0$ then we know that $f\in\IdlGen{P}$ and otherwise we know $f\notin\IdlGen{P}$.

In the rest of \Cref{sec:polyfin} we describe the three facets of this procedure.
In particular, \Cref{sec:muldivfin,sec:grfin,sec:buchfin} respectively discuss 
the multivariate division algorithm, Gr\"{o}bner bases, and Buchberger's algorithm.
%
\subsubsection{Multivariate division algorithm}\label{sec:muldivfin}
%
Let's first discuss the division algorithm for univariate polynomials.
%
A single step of division of a polynomial $f\in\poly{\K}{\{x\}}$ with a set of polynomials $P\subseteq\poly{\K}{\{x\}}$ goes as follows:
pick $p$ in $P$ whose degree is not larger than $f$,
multiply $p$ with $c\cdot x^m$ for some $c\in\K$ and $m\in\N$ such that $f$ and $cx^m p$ have the same degree and leading coefficient (coefficient of the highest degree term),
subtract $cx^mp$ from $f$ to get the polynomial $f' = f - cx^mp$ whose degree is smaller than $f$.
We continue this process until the degree of the resulting polynomial is smaller than the degree of all the polynomials in $P$.
%
\begin{example}
Let $f = x^5$ and $P = \{x^5 + 1, x^3 + x^2\}$.
We can have the following sequence of division steps.
\[
x^7 - 3x^4\xrightarrow{-x^2(x^5 + 1)}
-3x^4 - x^2 \xrightarrow{+3x(x^3 + x^2)}
3x^3 - x^2 \xrightarrow{-3(x^3 + x^2)}
-4x^2 \ .
\]
\end{example}
%
For univariate polynomials the division algorithm terminates since every step of the algorithm decreases the degree of the polynomial being divided.

Let $\mon{X}$ denote the set of monomials in $X$.
We extend the division algorithm to multivariate polynomials we have to first define a \kl{well-ordering} on the monomials.
Given such an well-ordering $\sqsubseteq$,
the \intro{leading monomial} of $f\in\poly{\K}{X}$ (denoted as \intro{$\lm[\sqsubseteq]{(f)}$}) is the largest monomial appearing in $f$.
The \intro{leading coefficient} of $f$ (denoted as \intro{$\lc[\sqsubseteq]{(f)}$}) is the coefficient of $\lm[\sqsubseteq]{(f)}$ in $f$.
The \intro{leading term} of $f$ (denoted as \intro{$\lt[\sqsubseteq]{(f)}$}) is defined as the product $\lc[\sqsubseteq]{(f)}\cdot\lm[\sqsubseteq]{(f)}$ in $f$.
Now a single step of division of a polynomial $f$ by a set of polynomials $P$ is as follows:
pick $p$ in $P$ such that $\lm[\sqsubseteq]{(p)}$ divides $\lm[\sqsubseteq]{(f)}$,
subtract $\frac{\lt[<]{(f)}}{\lt[<]{(p)}}\cdot p$ from $f$.
Note that there can be more than one choice for $p$,
therefore the division algorithm is non-deterministic.

We write $f \xrightarrow[P]{} f'$ to denote that $f'$ is the result of one step of division of $f$ by $P$.
Since the choice of $p$ from $P$ in the division step is non-deterministic,
$\xrightarrow[P]{}$ defines a binary relation (and not a function) on $\poly{\K}{X}$.
We use $\xrightarrow[P]{}^*$ to denote its transitive closure.
%
\begin{example}
Let $X = \{x_1,\dots,x_n\}$ and let $\lexlt$ denote the lexicographic order on $\mon{X}$.
That is,
\[
x_1^{m_1}\cdot\dots\cdot x_k^{m_k} \lexlt
x_1^{n_1}\cdot\dots\cdot x_k^{n_k}
\]
if there exists $\ell \leq k$ such that $m_{\ell} < n_{\ell}$ and $m_i = n_i$ for all $i < \ell$.
Let $f = 2x_1^2x_2 - x^5_3$ and $P = \{x_1x_2 + x_4, x_1^2 - x_3\}$.
Taking $p = x_1x_2 + x_4$ we get
\[
2x_1^2x_2 - x_1x^5_3 - 2x_1(x_1x_2 + x_4) = - 2x_1x_4 - x_1x^5_3\ .
\]
\end{example}
%
Observe that $\lm[\lexlt]{(- 2x_1x_4 - x_1x^5_3)} = x_1x_4 \lexlt x_1^2x_2 = \lm[\lexlt]{(2x_1^2x_2 - x^5_3)}$.
In fact, to guarantee that every run of the division algorithm terminates,
we need that for $p,f\in\poly{\K}{X}$,
if $\lm[\sqsubseteq]{(p)}$ divides $\lm[\sqsubseteq]{(f)}$ then
\[
\lm[\sqsubseteq]{\left(f - \frac{\lt[<]{(f)}}{\lt[<]{(p)}}\cdot p\right)}
\sqsubset \lm[\sqsubseteq]{(f)} \ .
\]
This is not guaranteed for any arbitrary well-ordering of $\mon{X}$,
but is true when the well-order is a term order.

A \kl{well-order} $\sqsubseteq$ on $\mon{X}$ is called a \intro{term order} if
\begin{enumerate}
\item $1 \sqsubseteq x$ for every $x\in X$, and
\item if $\monelt[m]\monelt[p] \sqsubseteq \monelt[m]\monelt[q]$ for every $\monelt[m]\in\mon{X}$ and $\monelt[p] \sqsubseteq \monelt[q] \in \mon{X}$. 
\end{enumerate} 
%
\arka{possible repetition}
%
\begin{lemma}
Let $\sqsubseteq$ be a term-order on $\mon{X}$ and $f,p\in\poly{\K}{X}$ be such that $\lm[\sqsubseteq]{p}$ divides $\lm[<]{(f)}$.
Then
\[
\lm[\sqsubseteq]{\left(f - \frac{\lt[<]{(f)}}{\lt[\sqsubseteq]{(p)}}\cdot p  \right)} < \lm[\sqsubseteq]{(f)} \ .
\]
\end{lemma}
%
\begin{example}[Term orderings]
For $\monelt[p]\in\mon{X}$ and $x\in X$,
let $\monelt{p}(x)$ denote the exponent of $x$ in $\monelt[p]$.

Observe that if $f \xrightarrow[P]{} f'$ then $f - f' \in \IdlGen{P}$.
Since ideals are closed under addition we get:
%
\begin{lemma}\label{lem:fin div}
If $f \xrightarrow[P]{}^* f'$ then $f - f' \in \IdlGen{P}$.
In particular, if $f \xrightarrow[P]{}^* 0$ then $f\in \IdlGen{P}$.
\end{lemma}
%
The above lemma gives a sufficient condition for a polynomial $f$ to be in $\IdlGen{P}$.
As the next example illustrates, the condition is not a necessary condition.
However, as we will see in \Cref{sec:grfin},
the condition becomes both necessary and sufficient when $P$ is a Gr\"{o}bner basis.
%
\begin{example}
Let $X = \{x < y\}$, $P = \{x + x^2y, x + xy^2\} \subseteq \poly{\Q}{X}$ and $f = x^3 + x \in \poly{\Q}{X}$.
We have
\[
f = x^3 + x = (x^2 + 1)(x + xy^2)  - xy(x + x^2y) \in \IdlGen{P}\ .
\]
But none of the elements in $\lm[\lexlt]{P} = \{x^2y,xy^2\}$ divides $\lm[\lexlt]{f} = x^3$,
which means $f$ can not be divided further using $P$ and  $f \xrightarrow[P]{}^* 0$ does not hold.
\end{example}

\subsubsection{Gr\"{o}bner bases}\label{sec:grfin}
%
A \intro{Gr\"{o}bner basis} for an ideal $\idl[I]$ with respect to a term order $\sqsubset$ is a subset $B\subseteq\idl[I]$ such that for every $f\in\idl[I]$ there exists $b\in B$ such that $\lm[\sqsubset]{b}$ divides $\lm[\sqsubset]{f}$.

For the remainder of \Cref{sec:grfin} we fix a term order $\sqsubseteq$.
%
\begin{lemma}\label{lem:grfin good}
If $B$ is a \intro{Gr\"{o}bner basis} for an ideal $\idl[I]$ then $f \xrightarrow[B]{}^* 0$ for every $f \in\idl[I]$.
\end{lemma}
%
A polynomial $f\in\poly{\K}{X}$ is said to be \intro{reduced} with respect to $H\subseteq\poly{\K}{X}$ if $f$ can not be divided further using $H$,
i.e.\
there is no $h\in H$ such that $\lm[\sqsubseteq]{h}$ divides $\lm[\sqsubseteq]{f}$.
A polynomial $f'\in\poly{\K}{X}$ is said to be a \intro{remainder} after dividing $f$ using some $P\subseteqfin \poly{\K}{X}$ if $f\xrightarrow[P]{}^* f'$ and $f'$ is reduced.
%
\begin{remark}
There can be more than one possible remainder.
For instance, let $f = xy$ and $P = \{x - 1,y - 1\}$.
Then both $y$ and $x$ are remainders after dividing $f$ using $P$.
If $P$ is a Gr\"{o}bner basis it is possible to modify the division algorithm to get a unique remainder.
We skip discussion of this modification since it is unnecessary for our purposes.
The interested reader is suggested to look at Sections 2.3 and 2.6 of \cite{CLO15}.
\end{remark}
%
\begin{corollary}
If $B$ is a \intro{Gr\"{o}bner basis} for an ideal $\idl[I]$ then $\idl[I] = \IdlGen{B}$,
and for every polynomial $f\in\poly{\K}{X}$ if $f'$ is a remainder of division of $f$ using $B$ then $f\in\idl[I]$ if and only if $f' = 0$.
\end{corollary}
%
The above corollary says that the multivariate division algorithm can be used to decide the membership problem for $\IdlGen{B}$ when $B$ is \kl{Gr\"{o}bner basis}.
Note that the multivariate division algorithm as defined is non-deterministic.
Below we give a determinisation of it.
%
\begin{algorithm}
    \label{alg:det div}
    \KwIn{A polynomial $f\in\poly{\K}{X}$ and a finite list of polynomials $P = [p_1,\dots,p_n]\subseteqfin \poly{\K}{X}$.}
    \KwOut{A \kl{remainder} of division of $f$ using $P$.}
    \Begin{
        $f' \gets f$\;
        \While{
        $\lm[\sqsubset](f')$ is divisible by some monomial in $\lm[\sqsubset](P)$
        }
        {
        $j \gets \min\setof{i}{\lm[\sqsubset](p_i)\text{ divides }\lm[\sqsubset](f')}$\;
        $f' = f' - \frac{\lt[\sqsubset](f')}{\lt[\sqsubset](p_j)}p_j$
        }
        \Return{$f'$}\;
    }
\end{algorithm}

\end{enumerate}
